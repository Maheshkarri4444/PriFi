STEP 1 — Download PTAU (ONLY ONCE)
    mkdir ptau

Download:
    curl -L \
    https://storage.googleapis.com/zkevm/ptau/powersOfTau28_hez_final_17.ptau \
    -o ptau/powersOfTau28_hez_final_17.ptau

DEPOSIT CIRCUIT

STEP 2 — Compile Deposit Circuit
    circom circuits/deposit_proof.circom \
        --r1cs \
        --wasm \
        --sym \
        -o build

STEP 3 — Generate Deposit ZKey
    snarkjs groth16 setup \
        build/deposit_proof.r1cs \
        ptau/powersOfTau28_hez_final_17.ptau \
        build/deposit_0000.zkey

STEP 4 — Finalize Deposit ZKey
    snarkjs zkey contribute \
        build/deposit_0000.zkey \
        build/deposit_final.zkey \
        --name="mahesh"

STEP 5 — Export Deposit Verifier
    snarkjs zkey export solidityverifier \
        build/deposit_final.zkey \
        contracts/DepositVerifier.sol

DONE.
